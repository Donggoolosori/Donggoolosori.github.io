# 문제
[10830 행렬 제곱](https://www.acmicpc.net/problem/10830)
# 풀이
우선 행렬끼리의 곱을 연산자 오버로딩으로 선언하여 코드의 가독성을 높였다.
```c++
typedef long long ll;
typedef vector<vector<ll>> matrix;

matrix operator*(const matrix &a, const matrix &b)
{
    matrix res(n, vector<ll>(n));

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            for (int k = 0; k < n; k++)
            {
                res[i][j] += a[i][k] * b[k][j];
            }
            res[i][j] %= 1000;
        }
    }
    return res;
}
```
이렇게 행렬의 곱을 연산자 오버로딩 해주면, 숫자를 제곱할 때와 동일하게 코드를 작성하여 제곱을 수행 할 수 있다.
```c++
// 행렬 a의 r제곱을 반환하는 함수
matrix power(matrix a, ll r)
{
    matrix res(n, vector<ll>(n));
    // res를 단위행렬로 설정
    for (int i = 0; i < n; i++)
        res[i][i] = 1;
    while (r > 0)
    {
        if (r % 2 == 1)
        {
            res = res * a;
        }
        r /= 2;
        a = a * a;
    }
    return res;
}
```

# 소스 코드
```c++
#include <iostream>
#include <vector>

using namespace std;
typedef long long ll;
typedef vector<vector<ll>> matrix;

ll n, b;

matrix operator*(const matrix &a, const matrix &b)
{
    matrix res(n, vector<ll>(n));

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            for (int k = 0; k < n; k++)
            {
                res[i][j] += a[i][k] * b[k][j];
            }
            res[i][j] %= 1000;
        }
    }
    return res;
}

// 행렬 a의 r제곱을 반환하는 함수
matrix power(matrix a, ll r)
{
    matrix res(n, vector<ll>(n));
    // res를 단위행렬로 설정
    for (int i = 0; i < n; i++)
        res[i][i] = 1;
    while (r > 0)
    {
        if (r % 2 == 1)
        {
            res = res * a;
        }
        r /= 2;
        a = a * a;
    }
    return res;
}

// 결과 행렬 출력
void printRes(const matrix &res)
{
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            cout << res[i][j] << " ";
        }
        cout << '\n';
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    cin >> n >> b;
    matrix origin(n, vector<ll>(n));

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            cin >> origin[i][j];
        }
    }

    printRes(power(origin, b));

    return 0;
}
```
